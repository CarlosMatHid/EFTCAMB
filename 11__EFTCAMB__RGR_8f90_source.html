67  subroutine eftcambreturntogr()
68 
69  implicit none
70 
71  real(dl) :: eft_functions_t1(21), eft_functions_t2(21)
72  real(dl) :: a_initial, a_final, log_a_initial, log_a_final, log_a_used, a_used
73  real(dl) :: last_a
74  integer :: i
75 
76  ! initial and final scale factors:
77  a_initial = cp%EFTCAMB%EFTCAMB_turn_on_time
78  a_final = 1._dl
79  ! convert to logarithm of the scale factor:
80  log_a_initial = log10( a_initial )
81  log_a_final = log10( a_final )
82  ! compute RGR functions at initial time:
83  log_a_used = log_a_initial
84  a_used = 10._dl**log_a_used
85  eft_functions_t1 = 0._dl
86  eft_functions_t2 = 0._dl
87  call eftcambreturntogr_functions( a_used, eft_functions_t2 )
88  last_a = a_used
89 
90  ! check if already above threshold:
91  if ( any( eft_functions_t2 > 0._dl ) ) then
92  cp%EFTCAMB%EFTCAMB_turn_on_time = last_a
93  call eftcambreturntogr_feedback()
94  return
95  end if
96 
97  ! loop over the logarithm of the scale factor:
98  do i=2, eft_rgr_num_points
99  ! initialize:
100  log_a_used = log_a_initial +real( i-1 )/real( eft_rgr_num_points -1 )*( log_a_final -log_a_initial )
101  a_used = 10._dl**log_a_used
102  eft_functions_t1 = 0._dl
103  ! compute RGR functions:
104  call eftcambreturntogr_functions( a_used, eft_functions_t1 )
105 
106  ! check if above threshold:
107  if ( any( eft_functions_t1 > 0._dl ) ) then
108  cp%EFTCAMB%EFTCAMB_turn_on_time = last_a
109  call eftcambreturntogr_feedback()
110  return
111  end if
112 
113  ! check if the threshold has been crossed:
114  eft_functions_t2 = eft_functions_t1*eft_functions_t2
115  if ( any( eft_functions_t2 < 0._dl ) ) then
116  cp%EFTCAMB%EFTCAMB_turn_on_time = last_a
117  call eftcambreturntogr_feedback()
118  return
119  else
120  ! prepare everything to proceed to the next step:
121  last_a = a_used
122  eft_functions_t2 = eft_functions_t1
123  end if
124  end do
125 
126  call eftcambreturntogr_feedback()
127 
128  end subroutine eftcambreturntogr

132  subroutine eftcambreturntogr_functions( a, eft_functions )
133 
134  implicit none
135 
136  real(dl), intent(in) :: a
137  real(dl), intent(out) :: eft_functions(21)
138 
139  type(eftcamb_timestep_cache) :: eft_cache
140  real(dl) :: a2, adotoa, EFT_grhonu, EFT_gpinu, EFT_grhonudot, EFT_gpinudot, grhormass_t
141  integer :: nu_i
142 
143  ! prepare:
144  a2 = a*a
145  ! initialize the cache:
146  call eft_cache%initialize()
147  ! start filling:
148  eft_cache%a = a
149  ! compute background densities of different species
150  eft_cache%grhob_t = grhob/a ! 8\pi G_N \rho_b a^2: baryons background density
151  eft_cache%grhoc_t = grhoc/a ! 8\pi G_N \rho_{cdm} a^2: cold dark matter background density
152  eft_cache%grhor_t = grhornomass/a2 ! 8\pi G_N \rho_{\nu} a^2: massless neutrinos background density
153  eft_cache%grhog_t = grhog/a2 ! 8\pi G_N \rho_{\gamma} a^2: radiation background density
154  ! Massive neutrinos terms:
155  if ( cp%Num_Nu_Massive /= 0 ) then
156  do nu_i = 1, cp%Nu_mass_eigenstates
157  eft_grhonu = 0._dl
158  eft_gpinu = 0._dl
159  grhormass_t=grhormass(nu_i)/a**2
160  call nu_background(a*nu_masses(nu_i),eft_grhonu,eft_gpinu)
161  eft_cache%grhonu_tot = eft_cache%grhonu_tot + grhormass_t*eft_grhonu
162  eft_cache%gpinu_tot = eft_cache%gpinu_tot + grhormass_t*eft_gpinu
163  end do
164  end if
165  ! assemble total densities and pressure:
166  eft_cache%grhom_t = eft_cache%grhob_t +eft_cache%grhoc_t +eft_cache%grhor_t +eft_cache%grhog_t +eft_cache%grhonu_tot
167  eft_cache%gpresm_t = (+eft_cache%grhor_t +eft_cache%grhog_t)/3._dl +eft_cache%gpinu_tot
168  ! compute the other things:
169  select type ( model => cp%EFTCAMB%model )
170  ! compute the background and the background EFT functions.
171  class is ( eftcamb_full_model )
172  ! background for full models. Here the expansion history is computed from the
173  ! EFT functions. Hence compute them first and then compute the expansion history.
174  call cp%EFTCAMB%model%compute_background_EFT_functions( a, cp%eft_par_cache , eft_cache )
175  call cp%EFTCAMB%model%compute_adotoa( a, cp%eft_par_cache , eft_cache )
176  class is ( eftcamb_designer_model )
177  ! background for designer models. Here the expansion history is parametrized
178  ! and does not depend on the EFT functions. Hence compute first the expansion history
179  ! and then the EFT functions.
180  call cp%EFTCAMB%model%compute_adotoa( a, cp%eft_par_cache , eft_cache )
181  end select
182  ! store adotoa:
183  adotoa = eft_cache%adotoa
184  ! compute massive neutrinos stuff:
185  ! Massive neutrinos mod:
186  eft_cache%grhonu_tot = 0._dl
187  eft_cache%gpinu_tot = 0._dl
188  if ( cp%Num_Nu_Massive /= 0 ) then
189  do nu_i = 1, cp%Nu_mass_eigenstates
190  eft_grhonu = 0._dl
191  eft_gpinu = 0._dl
192  eft_grhonudot = 0._dl
193  eft_gpinudot = 0._dl
194  grhormass_t=grhormass(nu_i)/a**2
195  call nu_background(a*nu_masses(nu_i),eft_grhonu,eft_gpinu)
196  eft_cache%grhonu_tot = eft_cache%grhonu_tot + grhormass_t*eft_grhonu
197  eft_cache%gpinu_tot = eft_cache%gpinu_tot + grhormass_t*eft_gpinu
198  eft_cache%grhonudot_tot = eft_cache%grhonudot_tot + grhormass_t*(nu_drho(a*nu_masses(nu_i) ,adotoa, eft_grhonu)&
199  & -4._dl*adotoa*eft_grhonu)
200  eft_cache%gpinudot_tot = eft_cache%gpinudot_tot + grhormass_t*(nu_pidot(a*nu_masses(nu_i),adotoa, eft_gpinu )&
201  & -4._dl*adotoa*eft_gpinu)
202  end do
203  end if
204  ! compute pressure dot:
205  eft_cache%gpresdotm_t = -4._dl*adotoa*( eft_cache%grhog_t +eft_cache%grhor_t )/3._dl +eft_cache%gpinudot_tot
206  ! compute remaining quantities related to H:
207  call cp%EFTCAMB%model%compute_H_derivs( a, cp%eft_par_cache , eft_cache )
208  ! compute backgrond EFT functions if model is designer:
209  select type ( model => cp%EFTCAMB%model )
210  class is ( eftcamb_designer_model )
211  call cp%EFTCAMB%model%compute_background_EFT_functions( a, cp%eft_par_cache , eft_cache )
212  end select
213  ! compute all other background stuff:
214  call cp%EFTCAMB%model%compute_rhoQPQ( a, cp%eft_par_cache , eft_cache )
215  ! compute second order EFT functions:
216  call cp%EFTCAMB%model%compute_secondorder_EFT_functions( a, cp%eft_par_cache , eft_cache )
217 
218  ! get the EFT functions:
219  eft_functions( 1) = abs( eft_cache%EFTOmegaV )
220  eft_functions( 2) = abs( a*adotoa*eft_cache%EFTOmegaP )
221  eft_functions( 3) = 0._dl
222  eft_functions( 4) = 0._dl
223  eft_functions( 5) = abs( eft_cache%EFTc/a2 )
224  eft_functions( 6) = abs( eft_cache%EFTLambda/a2 +cp%eft_par_cache%grhov )
225  eft_functions( 7) = abs( eft_cache%EFTcdot/a2 )
226  eft_functions( 8) = abs( eft_cache%EFTLambdadot/a2 )
227  eft_functions( 9) = abs( eft_cache%EFTGamma1V )
228  eft_functions(10) = abs( eft_cache%EFTGamma1P )
229  eft_functions(11) = abs( eft_cache%EFTGamma2V )
230  eft_functions(12) = abs( eft_cache%EFTGamma2P )
231  eft_functions(13) = abs( eft_cache%EFTGamma3V )
232  eft_functions(14) = abs( eft_cache%EFTGamma3P )
233  eft_functions(15) = abs( eft_cache%EFTGamma4V )
234  eft_functions(16) = abs( eft_cache%EFTGamma4P )
235  eft_functions(17) = 0._dl
236  eft_functions(18) = abs( eft_cache%EFTGamma5V )
237  eft_functions(19) = abs( eft_cache%EFTGamma5P )
238  eft_functions(20) = abs( eft_cache%EFTGamma6V )
239  eft_functions(21) = abs( eft_cache%EFTGamma6P )
240 
241  ! subtract the GR threshold:
242  eft_functions = eft_functions -efttogr
243 
244  end subroutine eftcambreturntogr_functions
