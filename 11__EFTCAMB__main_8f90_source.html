236  write(*,"(A24,I3)") ' FullMappingEFTmodel =', self%FullMappingEFTmodel

247  subroutine allocate_eftcamb_model( self )
248 
249  implicit none
250 
251  class(eftcamb) :: self
252 
253  ! check the allocation of the model:
254  if ( allocated(self%model) ) deallocate(self%model)
255 
256  ! do the allocation:
257  select case ( self%EFTflag )
258 
259  case (0) ! GR: no need to allocate
260 
261  case (1) ! Pure EFT:
262 
263  select case ( self%PureEFTmodel )
264  case(1)
265  allocate( eftcamb_std_pure_eft::self%model )
266  call self%model%init( 'Standard Pure EFT', 'Standard Pure EFT' )
267  case default
268  write(*,'(a,I3)') 'No model corresponding to EFTFlag =', self%EFTflag
269  write(*,'(a,I3)') 'and PureEFTmodel =', self%PureEFTmodel
270  write(*,'(a)') 'Please select an appropriate model:'
271  write(*,'(a)') 'PureEFTmodel=1 standard Pure EFT'
272  call mpistop('EFTCAMB error')
273  end select
274 
275  case (2) ! Alternative EFT:
276 
277  select case ( self%AltParEFTmodel )
278  case(1)
279  allocate( eftcamb_rph::self%model )
280  call self%model%init( 'RPH', 'RPH' )
281  case default
282  write(*,'(a,I3)') 'No model corresponding to EFTFlag =', self%EFTflag
283  write(*,'(a,I3)') 'and AltParEFTmodel =', self%AltParEFTmodel
284  write(*,'(a)') 'Please select an appropriate model:'
285  write(*,'(a)') 'AltParEFTmodel=1 reparametrized Horndeski'
286  call mpistop('EFTCAMB error')
287  end select
288 
289  case (3) ! Designer mapping EFT:
290 
291  select case ( self%DesignerEFTmodel )
292  case(1)
293  allocate( eftcamb_fr_designer::self%model )
294  call self%model%init( 'Designer f(R)', 'Designer f(R)' )
295  case(2)
296  allocate( eftcamb_des_mc_quint::self%model )
297  call self%model%init( 'Designer minimally coupled quintessence', 'Designer minimally coupled quintessence' )
298  case default
299  write(*,'(a,I3)') 'No model corresponding to EFTFlag =', self%EFTflag
300  write(*,'(a,I3)') 'and DesignerEFTmodel =', self%DesignerEFTmodel
301  write(*,'(a)') 'Please select an appropriate model:'
302  write(*,'(a)') 'DesignerEFTmodel=1 designer f(R)'
303  write(*,'(a)') 'DesignerEFTmodel=2 designer minimally coupled quintessence'
304  call mpistop('EFTCAMB error')
305  end select
306 
307  case (4) ! Full mapping EFT:
308 
309  select case ( self%FullMappingEFTmodel )
310  case(1)
311  allocate( eftcamb_horava::self%model )
312  call self%model%init( 'Horava', 'Horava' )
313  case default
314  write(*,'(a,I3)') 'No model corresponding to EFTFlag =', self%EFTflag
315  write(*,'(a,I3)') 'and FullMappingEFTmodel =', self%FullMappingEFTmodel
316  write(*,'(a)') 'Please select an appropriate model:'
317  write(*,'(a)') 'FullMappingEFTmodel=1 Horava gravity'
318  call mpistop('EFTCAMB error')
319  end select
320 
321  case default ! not found:
322 
323  write(*,'(a,I3)') 'No model corresponding to EFTFlag =', self%EFTflag
324  write(*,'(a)') 'Please select an appropriate model:'
325  write(*,'(a)') 'EFTFlag=0 GR code'
326  write(*,'(a)') 'EFTFlag=1 Pure EFT'
327  write(*,'(a)') 'EFTFlag=2 EFT alternative parametrizations'
328  write(*,'(a)') 'EFTFlag=3 designer mapping EFT'
329  write(*,'(a)') 'EFTFlag=4 full mapping EFT'
330  call mpistop('EFTCAMB error')
331 
332  end select
333 
334  end subroutine allocate_eftcamb_model
